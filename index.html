  function removeDuplicateWords(s) {
      let arr = s.split(" ");
      let set = new Set();

      for (const item of arr) {
         set.add(item);
      }
      arr = [];
      for (const kei of set) {
       arr.push(kei);
      }
      return arr.join(" ");
      }
